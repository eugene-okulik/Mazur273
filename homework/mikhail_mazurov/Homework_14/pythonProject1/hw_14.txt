FROM students s 
JOIN `groups` g
ON s.group_id = g.id
JOIN books b
ON s.id = b.taken_by_student_id 
JOIN marks m 
ON s.id = m.student_id 
JOIN lessons l 
ON m.lesson_id = l.id
JOIN subjects sub
ON l.subject_id = sub.id 
WHERE s.id = 22898;
